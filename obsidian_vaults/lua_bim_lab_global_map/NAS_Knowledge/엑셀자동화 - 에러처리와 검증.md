## 1. 파일 손상 방지 패턴

### 원자적 저장 (Atomic Save)

파일 저장 중 실패 시 원본 파일 손상을 막는다.

```python
import shutil
import tempfile
from pathlib import Path

def safe_save(wb, target_path: str) -> None:
    target = Path(target_path)
    # 임시 파일에 저장 후 원본으로 이동
    with tempfile.NamedTemporaryFile(
        suffix=".xlsx", dir=target.parent, delete=False
    ) as tmp:
        tmp_path = Path(tmp.name)
    try:
        wb.save(str(tmp_path))
        shutil.move(str(tmp_path), str(target))
    except Exception:
        tmp_path.unlink(missing_ok=True)
        raise
```

### 백업 후 덮어쓰기

```python
from datetime import datetime

def save_with_backup(wb, target_path: str, keep_backups: int = 3) -> None:
    target = Path(target_path)
    if target.exists():
        ts = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup = target.with_stem(f"{target.stem}_bak_{ts}")
        shutil.copy2(target, backup)

        # 오래된 백업 삭제
        backups = sorted(
            target.parent.glob(f"{target.stem}_bak_*.xlsx"),
            key=lambda p: p.stat().st_mtime, reverse=True
        )
        for old in backups[keep_backups:]:
            old.unlink()

    safe_save(wb, target_path)
```

## 2. 인코딩 안전 읽기

```python
import csv
import chardet

def read_csv_safe(path: str, detect_encoding: bool = False) -> list[dict]:
    """
    1차: 고정 후보 순서 시도
    2차: chardet 자동 감지 (detect_encoding=True 시)
    """
    candidates = ["utf-8-sig", "cp949", "utf-8", "euc-kr"]

    for enc in candidates:
        try:
            with open(path, newline="", encoding=enc) as f:
                rows = list(csv.DictReader(f))
            # 한글 깨짐 감지: 대표 값에 replacement character 포함 여부
            sample = str(rows[:3]) if rows else ""
            if "�" not in sample:
                return rows
        except (UnicodeDecodeError, LookupError):
            continue

    if detect_encoding:
        with open(path, "rb") as f:
            raw = f.read(50000)
        detected = chardet.detect(raw)
        enc = detected.get("encoding") or "utf-8"
        with open(path, newline="", encoding=enc, errors="replace") as f:
            return list(csv.DictReader(f))

    raise ValueError(f"인코딩 감지 실패: {path}")


def read_excel_safe(path: str, sheet_name: str = None) -> list[dict]:
    from openpyxl import load_workbook
    try:
        wb = load_workbook(path, data_only=True, read_only=True)
        ws = wb[sheet_name] if sheet_name and sheet_name in wb.sheetnames else wb.active
        rows = list(ws.iter_rows(values_only=True))
        wb.close()
    except Exception as e:
        raise ValueError(f"Excel 읽기 실패 ({path}): {e}") from e

    if not rows:
        return []
    headers = [str(h) if h is not None else f"col_{i}" for i, h in enumerate(rows[0])]
    return [dict(zip(headers, row)) for row in rows[1:] if any(v is not None for v in row)]
```

## 3. 메모리 안전 (대용량 처리)

```python
from openpyxl import Workbook

def stream_to_excel(data_gen, headers: list[str], output_path: str) -> int:
    """
    write_only 모드: 5만 행 이상 처리용
    반환값: 저장된 행 수
    주의: write_only 모드에서는 셀 서식/수식/테이블 불가
    """
    wb = Workbook(write_only=True)
    ws = wb.create_sheet()
    ws.append(headers)
    count = 0
    for row in data_gen:
        ws.append(row)
        count += 1
    wb.save(output_path)
    return count


def chunk_excel_by_rows(
    data: list[list],
    headers: list[str],
    output_dir: str,
    chunk_size: int = 50000,
    base_name: str = "output",
) -> list[str]:
    """5MB 초과 우려 시 시트/파일 분할"""
    from math import ceil
    output_paths = []
    total_chunks = ceil(len(data) / chunk_size)

    for i in range(total_chunks):
        chunk = data[i * chunk_size:(i + 1) * chunk_size]
        path = f"{output_dir}/{base_name}_part{i+1:02d}.xlsx"
        wb = Workbook(write_only=True)
        ws = wb.create_sheet(f"데이터_{i+1}")
        ws.append(headers)
        for row in chunk:
            ws.append(row)
        wb.save(path)
        output_paths.append(path)

    return output_paths
```

## 4. 입력 데이터 검증

```python
from dataclasses import dataclass
from typing import Any

@dataclass
class ValidationError:
    row: int
    field: str
    value: Any
    reason: str

def validate_clash_data(rows: list[dict]) -> tuple[list[dict], list[ValidationError]]:
    """간섭 데이터 유효성 검증"""
    valid = []
    errors = []
    required = {"공종A", "공종B", "레벨", "상태"}
    valid_statuses = {"신규", "검토중", "조치완료", "보류", "제외"}

    for i, row in enumerate(rows, 2):  # 엑셀 행 번호 기준
        row_errors = []

        # 필수 컬럼
        for col in required:
            if not row.get(col, "").strip():
                row_errors.append(ValidationError(i, col, row.get(col), "필수값 누락"))

        # 상태값
        status = row.get("상태", "")
        if status and status not in valid_statuses:
            row_errors.append(ValidationError(i, "상태", status, f"허용값: {valid_statuses}"))

        # 간섭거리 숫자 확인
        dist = row.get("간섭거리(mm)", "")
        if dist:
            try:
                float(str(dist).replace(",", ""))
            except ValueError:
                row_errors.append(ValidationError(i, "간섭거리(mm)", dist, "숫자가 아님"))

        if row_errors:
            errors.extend(row_errors)
        else:
            valid.append(row)

    return valid, errors


def write_validation_report(errors: list[ValidationError], output_path: str) -> None:
    wb = Workbook()
    ws = wb.active
    ws.title = "검증오류"
    ws.append(["행", "필드", "값", "오류사유"])
    _apply_header_style(ws, row=1)
    for err in errors:
        ws.append([err.row, err.field, str(err.value), err.reason])
    ws.auto_filter.ref = f"A1:D{ws.max_row}"
    _auto_fit_columns(ws)
    wb.save(output_path)
```

## 5. 복구 패턴

```python
from contextlib import contextmanager
import logging

logger = logging.getLogger(__name__)

@contextmanager
def excel_transaction(output_path: str):
    """
    성공 시: 저장, 실패 시: 원본 보존 + 오류 로그
    """
    backup_path = None
    target = Path(output_path)
    if target.exists():
        backup_path = target.with_suffix(".bak.xlsx")
        shutil.copy2(target, backup_path)

    try:
        yield
        if backup_path and backup_path.exists():
            backup_path.unlink()
        logger.info(f"저장 성공: {output_path}")
    except Exception as e:
        logger.error(f"저장 실패, 원본 복원 중: {e}")
        if backup_path and backup_path.exists():
            shutil.move(str(backup_path), str(target))
        raise


# 사용 예시
def generate_report(data, output_path: str):
    with excel_transaction(output_path):
        wb = Workbook()
        ws = wb.active
        for row in data:
            ws.append(row)
        safe_save(wb, output_path)
```

## 6. 셀 값 타입 안전 변환

```python
from datetime import datetime, date
from typing import Any

def safe_cell_value(value: Any, expected_type: str = "str") -> Any:
    """openpyxl 셀 값을 안전하게 변환"""
    if value is None:
        return {"str": "", "int": 0, "float": 0.0, "date": None}.get(expected_type, None)

    if expected_type == "str":
        return str(value).strip()

    if expected_type == "int":
        try:
            return int(float(str(value).replace(",", "")))
        except (ValueError, TypeError):
            return 0

    if expected_type == "float":
        try:
            return float(str(value).replace(",", ""))
        except (ValueError, TypeError):
            return 0.0

    if expected_type == "date":
        if isinstance(value, (datetime, date)):
            return value if isinstance(value, date) else value.date()
        try:
            return datetime.strptime(str(value), "%Y-%m-%d").date()
        except ValueError:
            return None

    return value


def normalize_row(row: dict, schema: dict[str, str]) -> dict:
    """
    schema: {"컬럼명": "str|int|float|date"}
    """
    return {k: safe_cell_value(row.get(k), t) for k, t in schema.items()}
```

## 7. 공통 헬퍼

```python
def _apply_header_style(ws, row: int = 1) -> None:
    from openpyxl.styles import PatternFill, Font, Alignment, Border, Side
    fill = PatternFill(fill_type="solid", fgColor="1F4E79")
    font = Font(bold=True, color="FFFFFF", size=11, name="맑은 고딕")
    align = Alignment(horizontal="center", vertical="center", wrap_text=True)
    thin = Side(style="thin", color="FFFFFF")
    border = Border(left=thin, right=thin, top=thin, bottom=thin)
    for cell in ws[row]:
        cell.fill = fill
        cell.font = font
        cell.alignment = align
        cell.border = border
    ws.row_dimensions[row].height = 30

def _auto_fit_columns(ws, min_w: int = 8, max_w: int = 40) -> None:
    for col in ws.columns:
        letter = col[0].column_letter
        max_len = max((len(str(c.value)) for c in col if c.value), default=0)
        ws.column_dimensions[letter].width = min(max(max_len + 2, min_w), max_w)
```

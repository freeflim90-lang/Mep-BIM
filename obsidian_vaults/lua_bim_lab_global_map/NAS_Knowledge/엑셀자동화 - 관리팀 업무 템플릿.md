# 엑셀 자동화 — 관리팀 업무 템플릿

> **보안 규칙**: 개인정보(이름 제외), 급여, 계약금액 → `***` 마스킹 후 처리

---

## 1. 근태 취합 자동화

### 입력 데이터 형식 (CSV)

```
이름,날짜,출근시간,퇴근시간
홍길동,2026-05-01,09:05,18:30
```

### 근태 집계 Excel 생성

```python
import csv
from datetime import datetime, timedelta, date
from collections import defaultdict
from openpyxl import Workbook
from openpyxl.styles import PatternFill, Font, Alignment
from openpyxl.utils import get_column_letter
from openpyxl.worksheet.table import Table, TableStyleInfo
from openpyxl.formatting.rule import CellIsRule, FormulaRule

STANDARD_IN = datetime.strptime("09:00", "%H:%M")
STANDARD_OUT = datetime.strptime("18:00", "%H:%M")
LATE_THRESHOLD = timedelta(minutes=10)

def build_attendance_report(csv_path: str, output_path: str, year_month: str) -> None:
    records = _load_attendance_csv(csv_path)
    wb = Workbook()
    ws = wb.active
    ws.title = f"{year_month}_근태"

    headers = ["이름", "날짜", "출근", "퇴근", "근무시간", "지각", "조퇴", "비고"]
    ws.append(headers)
    _apply_header_style(ws, row=1)
    ws.freeze_panes = "A2"

    for rec in records:
        actual_in = datetime.strptime(rec["출근시간"], "%H:%M")
        actual_out = datetime.strptime(rec["퇴근시간"], "%H:%M")
        work_hours = (actual_out - actual_in).seconds / 3600
        late = "Y" if (actual_in - STANDARD_IN) > LATE_THRESHOLD else ""
        early_leave = "Y" if (STANDARD_OUT - actual_out) > LATE_THRESHOLD else ""

        ws.append([
            rec["이름"],
            rec["날짜"],
            rec["출근시간"],
            rec["퇴근시간"],
            round(work_hours, 2),
            late,
            early_leave,
            "",
        ])

    last_row = ws.max_row

    # 지각/조퇴 강조
    yellow = PatternFill(fill_type="solid", fgColor="FFD966")
    red = PatternFill(fill_type="solid", fgColor="FF9999")
    ws.conditional_formatting.add(
        f"F2:F{last_row}",
        CellIsRule(operator="equal", formula=['"Y"'], fill=yellow)
    )
    ws.conditional_formatting.add(
        f"G2:G{last_row}",
        CellIsRule(operator="equal", formula=['"Y"'], fill=red)
    )

    # 근무시간 서식
    for row in ws.iter_rows(min_row=2, max_row=last_row, min_col=5, max_col=5):
        for cell in row:
            cell.number_format = "#,##0.00"

    # 테이블
    last_col = get_column_letter(len(headers))
    table = Table(displayName="근태", ref=f"A1:{last_col}{last_row}")
    table.tableStyleInfo = TableStyleInfo(name="TableStyleMedium9", showRowStripes=True)
    ws.add_table(table)

    # 개인별 집계 시트
    _add_person_summary(wb, records, year_month)

    _auto_fit_columns(ws)
    wb.save(output_path)

def _load_attendance_csv(path: str) -> list[dict]:
    for enc in ("utf-8-sig", "cp949", "utf-8"):
        try:
            with open(path, newline="", encoding=enc) as f:
                return list(csv.DictReader(f))
        except (UnicodeDecodeError, LookupError):
            continue
    raise ValueError("인코딩 감지 실패")

def _add_person_summary(wb: Workbook, records: list[dict], year_month: str) -> None:
    ws = wb.create_sheet("개인별 집계")
    ws.append(["이름", "총 근무일", "지각 횟수", "조퇴 횟수", "평균 근무시간"])
    _apply_header_style(ws, row=1)

    from collections import defaultdict
    summary = defaultdict(lambda: {"days": 0, "late": 0, "early": 0, "hours": 0.0})
    for rec in records:
        name = rec["이름"]
        actual_in = datetime.strptime(rec["출근시간"], "%H:%M")
        actual_out = datetime.strptime(rec["퇴근시간"], "%H:%M")
        work_hours = (actual_out - actual_in).seconds / 3600
        summary[name]["days"] += 1
        summary[name]["hours"] += work_hours
        if (actual_in - STANDARD_IN) > LATE_THRESHOLD:
            summary[name]["late"] += 1
        if (STANDARD_OUT - actual_out) > LATE_THRESHOLD:
            summary[name]["early"] += 1

    for name, s in sorted(summary.items()):
        avg = s["hours"] / s["days"] if s["days"] else 0
        ws.append([name, s["days"], s["late"], s["early"], round(avg, 2)])

    ws.freeze_panes = "A2"
    _auto_fit_columns(ws)
```

## 2. 비용 정산 자동화

```python
from openpyxl.chart import BarChart, Reference

EXPENSE_CATEGORIES = ["인건비", "재료비", "외주비", "교통비", "기타"]

def build_expense_report(data: list[dict], output_path: str, month: str) -> None:
    """
    data: [{"날짜": "2026-05-03", "항목": "재료비", "금액": 150000, "내용": "...", "담당자": "..."}]
    """
    wb = Workbook()
    ws = wb.active
    ws.title = f"{month}_비용정산"

    headers = ["날짜", "항목", "금액", "내용", "담당자", "승인여부"]
    ws.append(headers)
    _apply_header_style(ws, row=1)
    ws.freeze_panes = "A2"

    for row in data:
        ws.append([
            row.get("날짜", ""),
            row.get("항목", ""),
            row.get("금액", 0),
            row.get("내용", ""),
            row.get("담당자", ""),
            row.get("승인여부", "대기"),
        ])

    last_row = ws.max_row

    # 금액 서식
    for row in ws.iter_rows(min_row=2, max_row=last_row, min_col=3, max_col=3):
        for cell in row:
            cell.number_format = '#,##0"원"'

    # 합계 행
    ws.append(["합계", "", f"=SUM(C2:C{last_row})", "", "", ""])
    sum_row = ws.max_row
    ws.cell(sum_row, 1).font = Font(bold=True)
    ws.cell(sum_row, 3).font = Font(bold=True)
    ws.cell(sum_row, 3).number_format = '#,##0"원"'

    # 예산 초과 강조 (500만원 기준)
    red = PatternFill(fill_type="solid", fgColor="FF9999")
    ws.conditional_formatting.add(
        f"C2:C{last_row}",
        CellIsRule(operator="greaterThan", formula=["5000000"], fill=red)
    )

    # 드롭다운
    from openpyxl.worksheet.datavalidation import DataValidation
    dv_cat = DataValidation(
        type="list",
        formula1=f'"{",".join(EXPENSE_CATEGORIES)}"',
        allow_blank=False
    )
    ws.add_data_validation(dv_cat)
    dv_cat.add(f"B2:B{last_row}")

    dv_status = DataValidation(
        type="list", formula1='"대기,승인,반려"', allow_blank=False
    )
    ws.add_data_validation(dv_status)
    dv_status.add(f"F2:F{last_row}")

    last_col = get_column_letter(len(headers))
    table = Table(displayName="비용정산", ref=f"A1:{last_col}{last_row}")
    table.tableStyleInfo = TableStyleInfo(name="TableStyleMedium9", showRowStripes=True)
    ws.add_table(table)

    # 항목별 피벗 시트
    _add_expense_pivot(wb, data, month)

    _auto_fit_columns(ws)
    wb.save(output_path)

def _add_expense_pivot(wb: Workbook, data: list[dict], month: str) -> None:
    ws = wb.create_sheet("항목별 집계")
    from collections import defaultdict
    pivot = defaultdict(float)
    for row in data:
        pivot[row.get("항목", "기타")] += float(row.get("금액", 0))

    ws.append(["항목", "합계(원)"])
    _apply_header_style(ws, row=1)
    for cat in EXPENSE_CATEGORIES:
        ws.append([cat, pivot.get(cat, 0)])
    ws.append(["총합", sum(pivot.values())])

    last_row = ws.max_row
    for row in ws.iter_rows(min_row=2, max_row=last_row, min_col=2, max_col=2):
        for cell in row:
            cell.number_format = '#,##0"원"'
    ws.cell(last_row, 1).font = Font(bold=True)
    ws.cell(last_row, 2).font = Font(bold=True)

    # 막대 차트
    chart = BarChart()
    chart.type = "col"
    chart.title = f"{month} 항목별 비용"
    chart.y_axis.title = "금액(원)"
    data_ref = Reference(ws, min_col=2, min_row=1, max_row=last_row - 1)
    cats = Reference(ws, min_col=1, min_row=2, max_row=last_row - 1)
    chart.add_data(data_ref, titles_from_data=True)
    chart.set_categories(cats)
    chart.style = 10
    ws.add_chart(chart, "D2")

    _auto_fit_columns(ws)
```

## 3. 구매 요청서 자동화

```python
def build_purchase_request(
    items: list[dict],
    requester: str,
    output_path: str,
    request_no: str,
) -> None:
    """
    items: [{"품목": "BIM 라이선스", "규격": "Revit 2026", "수량": 2, "단가": 1500000, "비고": ""}]
    """
    wb = Workbook()
    ws = wb.active
    ws.title = "구매요청서"

    # 상단 메타정보
    ws.merge_cells("A1:F1")
    ws["A1"] = "구매 요청서"
    ws["A1"].font = Font(bold=True, size=16)
    ws["A1"].alignment = Alignment(horizontal="center")

    ws["A2"] = "요청번호"
    ws["B2"] = request_no
    ws["D2"] = "요청자"
    ws["E2"] = requester
    ws["A3"] = "요청일"
    ws["B3"] = datetime.today().strftime("%Y-%m-%d")
    ws["D3"] = "부서"
    ws["E3"] = "관리팀"

    # 헤더
    headers = ["품목", "규격", "수량", "단가(원)", "합계(원)", "비고"]
    for i, h in enumerate(headers, 1):
        cell = ws.cell(5, i, h)
        cell.fill = PatternFill(fill_type="solid", fgColor="1F4E79")
        cell.font = Font(bold=True, color="FFFFFF", size=11)
        cell.alignment = Alignment(horizontal="center", vertical="center")

    # 품목 행
    for i, item in enumerate(items, 6):
        ws.cell(i, 1, item.get("품목", ""))
        ws.cell(i, 2, item.get("규격", ""))
        ws.cell(i, 3, item.get("수량", 1))
        ws.cell(i, 4, item.get("단가", 0))
        ws.cell(i, 5, f"=C{i}*D{i}")
        ws.cell(i, 6, item.get("비고", ""))
        ws.cell(i, 4).number_format = '#,##0"원"'
        ws.cell(i, 5).number_format = '#,##0"원"'

    last_item_row = 5 + len(items)

    # 합계
    total_row = last_item_row + 1
    ws.cell(total_row, 4, "총 합계")
    ws.cell(total_row, 4).font = Font(bold=True)
    ws.cell(total_row, 5, f"=SUM(E6:E{last_item_row})")
    ws.cell(total_row, 5).font = Font(bold=True)
    ws.cell(total_row, 5).number_format = '#,##0"원"'

    # 승인란
    sign_row = total_row + 2
    ws.cell(sign_row, 1, "담당")
    ws.cell(sign_row, 2, "팀장")
    ws.cell(sign_row, 3, "본부장")
    for col in range(1, 4):
        ws.cell(sign_row, col).alignment = Alignment(horizontal="center")

    ws.page_setup.paperSize = 9
    ws.page_setup.fitToPage = True
    ws.page_setup.fitToWidth = 1

    _auto_fit_columns(ws)
    wb.save(output_path)
```

## 4. 라이선스 관리 자동화

```python
from openpyxl.formatting.rule import FormulaRule

def build_license_tracker(licenses: list[dict], output_path: str) -> None:
    """
    licenses: [
        {"소프트웨어": "Revit 2026", "라이선스키": "***",
         "수량": 5, "구매일": "2025-01-01", "만료일": "2026-01-01",
         "담당자": "홍길동", "비용(원)": 7500000}
    ]
    """
    wb = Workbook()
    ws = wb.active
    ws.title = "라이선스 관리"

    headers = ["소프트웨어", "라이선스키", "수량", "구매일", "만료일", "D-Day", "담당자", "비용(원)", "상태"]
    ws.append(headers)
    _apply_header_style(ws, row=1)
    ws.freeze_panes = "A2"

    for lic in licenses:
        ws.append([
            lic.get("소프트웨어", ""),
            "***",                         # 라이선스키 마스킹
            lic.get("수량", 1),
            lic.get("구매일", ""),
            lic.get("만료일", ""),
            f'=E{ws.max_row+1}-TODAY()',   # D-Day 수식 (실제 행 번호로 치환)
            lic.get("담당자", ""),
            lic.get("비용(원)", 0),
            "",
        ])

    last_row = ws.max_row

    # D-Day 수식 (행 번호 보정)
    for i, row in enumerate(ws.iter_rows(min_row=2, max_row=last_row), 2):
        row[5].value = f"=E{i}-TODAY()"
        row[5].number_format = "0"

    # 날짜 서식
    for row in ws.iter_rows(min_row=2, max_row=last_row, min_col=4, max_col=5):
        for cell in row:
            cell.number_format = "YYYY-MM-DD"

    # 비용 서식
    for row in ws.iter_rows(min_row=2, max_row=last_row, min_col=8, max_col=8):
        for cell in row:
            cell.number_format = '#,##0"원"'

    # D-30 이내 만료 강조
    yellow = PatternFill(fill_type="solid", fgColor="FFD966")
    red = PatternFill(fill_type="solid", fgColor="FF9999")
    # 30일 이내
    ws.conditional_formatting.add(
        f"E2:E{last_row}",
        FormulaRule(formula=["AND(E2-TODAY()<=30,E2-TODAY()>=0)"], fill=yellow)
    )
    # 만료됨
    ws.conditional_formatting.add(
        f"E2:E{last_row}",
        FormulaRule(formula=["E2<TODAY()"], fill=red)
    )

    # 상태 자동 수식
    for i, row in enumerate(ws.iter_rows(min_row=2, max_row=last_row, min_col=9, max_col=9), 2):
        for cell in row:
            cell.value = f'=IF(E{i}<TODAY(),"만료",IF(E{i}-TODAY()<=30,"갱신필요","정상"))'

    last_col = get_column_letter(len(headers))
    table = Table(displayName="라이선스", ref=f"A1:{last_col}{last_row}")
    table.tableStyleInfo = TableStyleInfo(name="TableStyleMedium9", showRowStripes=True)
    ws.add_table(table)

    ws.print_title_rows = "1:1"
    _auto_fit_columns(ws)
    wb.save(output_path)
```

## 5. 월간 보고서 자동화

def build_monthly_report(
    month: str,                    # "2026-05"
    revenue: float,
    prev_revenue: float,
    expenses: dict[str, float],    # {"인건비": 5000000, ...}
    projects: list[dict],
    output_path: str,
) -> None:
    from openpyxl.chart import LineChart, Reference
    wb = Workbook()
    ws = wb.active
    ws.title = "월간보고서"

    # 제목
    ws.merge_cells("A1:F1")
    ws["A1"] = f"LUA BIM LABS — {month} 월간 보고서"
    ws["A1"].font = Font(bold=True, size=14)
    ws["A1"].alignment = Alignment(horizontal="center")
    ws.row_dimensions[1].height = 35

    # 매출/비용 요약
    ws["A3"] = "항목"
    ws["B3"] = "금액(원)"
    ws["C3"] = "전월(원)"
    ws["D3"] = "증감"
    ws["E3"] = "증감률"
    for col in range(1, 6):
        c = ws.cell(3, col)
        c.fill = PatternFill(fill_type="solid", fgColor="2E4057")
        c.font = Font(bold=True, color="FFFFFF")
        c.alignment = Alignment(horizontal="center")

    total_expense = sum(expenses.values())
    summary = [
        ("매출", revenue, prev_revenue),
        ("비용 합계", total_expense, None),
        ("영업이익", revenue - total_expense, None),
    ]
    for i, (label, amount, prev) in enumerate(summary, 4):
        ws.cell(i, 1, label)
        ws.cell(i, 2, amount).number_format = '#,##0"원"'
        if prev is not None:
            ws.cell(i, 3, prev).number_format = '#,##0"원"'
            ws.cell(i, 4, f"=B{i}-C{i}").number_format = '#,##0"원"'
            ws.cell(i, 5, f"=IFERROR((B{i}-C{i})/C{i},0)").number_format = "0.0%"

    # 비용 상세
    ws["A8"] = "비용 상세"
    ws["A8"].font = Font(bold=True)
    for j, (cat, amt) in enumerate(expenses.items(), 9):
        ws.cell(j, 1, cat)
        ws.cell(j, 2, amt).number_format = '#,##0"원"'

    # 진행 프로젝트
    proj_row = 9 + len(expenses) + 2
    ws.cell(proj_row, 1, "진행 프로젝트")
    ws.cell(proj_row, 1).font = Font(bold=True, size=11)
    ws.cell(proj_row + 1, 1, "프로젝트명")
    ws.cell(proj_row + 1, 2, "단계")
    ws.cell(proj_row + 1, 3, "완료율")
    ws.cell(proj_row + 1, 4, "담당자")
    for k, proj in enumerate(projects, proj_row + 2):
        ws.cell(k, 1, proj.get("name", ""))
        ws.cell(k, 2, proj.get("phase", ""))
        ws.cell(k, 3, proj.get("progress", 0)).number_format = "0%"
        ws.cell(k, 4, proj.get("manager", ""))

    _auto_fit_columns(ws)
    ws.page_setup.orientation = "portrait"
    ws.page_setup.paperSize = 9
    ws.page_setup.fitToPage = True
    ws.page_setup.fitToWidth = 1
    ws.oddHeader.center.text = f"LUA BIM LABS {month} 월간보고서"
    ws.oddFooter.right.text = "&P / &N"

    wb.save(output_path)

## 6. 공통 헬퍼 함수

```python
from openpyxl.styles import PatternFill, Font, Alignment, Border, Side

def _apply_header_style(ws, row: int = 1) -> None:
    fill = PatternFill(fill_type="solid", fgColor="1F4E79")
    font = Font(bold=True, color="FFFFFF", size=11, name="맑은 고딕")
    align = Alignment(horizontal="center", vertical="center", wrap_text=True)
    thin = Side(style="thin", color="FFFFFF")
    border = Border(left=thin, right=thin, top=thin, bottom=thin)
    for cell in ws[row]:
        cell.fill = fill
        cell.font = font
        cell.alignment = align
        cell.border = border
    ws.row_dimensions[row].height = 30

def _auto_fit_columns(ws, min_w: int = 8, max_w: int = 40) -> None:
    for col in ws.columns:
        letter = col[0].column_letter
        max_len = max((len(str(c.value)) for c in col if c.value), default=0)
        ws.column_dimensions[letter].width = min(max(max_len + 2, min_w), max_w)
```
